Task 1 — Power Function

Calculates base ^ exponent using both iterative and recursive JS functions.

MethodResult
Iterative (JS)
Recursive (JS)
View source code (PHP translation to JS) 
// Iterative approach
function powerIterative(base, exp) {
    let result = 1;
    for (let i = 0; i < Math.abs(exp); i++) result *= base;
    return exp < 0 ? 1 / result : result;
}

// Recursive approach
function powerRecursive(base, exp) {
    if (exp === 0) return 1;
    if (exp < 0) return 1 / powerRecursive(base, -exp);
    return base * powerRecursive(base, exp - 1);
}
